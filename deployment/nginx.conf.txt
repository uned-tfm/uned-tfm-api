upstream backend_hosts {
  server 127.0.0.1:8000;
}

client_max_body_size 5M;

server {
  listen 443 ssl;
  server_name app.acceptable.legal localhost;

  root /home/ubuntu/acceptable-apps/acceptable-webapp;
  index index.html;

  access_log /var/log/nginx/app-nock-payment.access.log;
  error_log /var/log/nginx/app-nock-payment.error.log debug;

  ssl_certificate /etc/letsencrypt/live/app.acceptable.legal/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/app.acceptable.legal/privkey.pem;

  location / {
    try_files $uri $uri/ /index.html;
  }

  location /malware/api {
    proxy_pass http://backend_hosts;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection 'upgrade';
    proxy_set_header Host $host;
    proxy_cache_bypass $http_upgrade;
  }
}

server {
  listen 80;
  server_name app.acceptable.legal localhost;
  return 301 https://app.acceptable.legal$request_uri;
}
